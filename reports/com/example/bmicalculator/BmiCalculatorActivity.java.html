1    package com.example.bmicalculator;
2    
3    import android.os.Bundle;
4    import android.widget.Button;
5    import android.widget.EditText;
6    import android.widget.TextView;
7    
8    import androidx.appcompat.app.AppCompatActivity;
9    
10   
11   public class BmiCalculatorActivity extends AppCompatActivity {
12   
13       EditText weightInput, heightInput;
14       Button calculateButton;
15       TextView resultText;
16   
17       @Override
18       protected void onCreate(Bundle savedInstanceState) {
19           super.onCreate(savedInstanceState);
20           setContentView(R.layout.activity_bmi_calculator);
21   
22           weightInput = findViewById(R.id.weightInput);
23           heightInput = findViewById(R.id.heightInput);
24           calculateButton = findViewById(R.id.calculateButton);
25           resultText = findViewById(R.id.resultText);
26   
27           calculateButton.setOnClickListener(v -> {
28               String weightStr = weightInput.getText().toString();
29               String heightStr = heightInput.getText().toString();
30   
31               if (!weightStr.isEmpty() && !heightStr.isEmpty()) {
32                   float weight = Float.parseFloat(weightStr);
33                   float height = Float.parseFloat(heightStr) / 100;
34   
35                   float bmi = weight / (height * height);
36                   String status;
37   
38                   if (bmi < 18.5)
39                       status = "Niedowaga";
40                   else if (bmi < 25)
41                       status = "Optimum";
42                   else if (bmi < 30)
43                       status = "Nadwaga";
44                   else
45                       status = "Otyłość";
46   
47                   resultText.setText(String.format("BMI: %.2f\nStatus: %s", bmi, status));
48               } else {
49                   resultText.setText("Wprowadź masę i wzrost.");
50               }
51           });
52       }
53   }
