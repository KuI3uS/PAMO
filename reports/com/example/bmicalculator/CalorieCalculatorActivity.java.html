1    package com.example.bmicalculator;
2    
3    import android.content.Intent;
4    import android.os.Bundle;
5    import android.widget.ArrayAdapter;
6    import android.widget.Button;
7    import android.widget.EditText;
8    import android.widget.RadioGroup;
9    import android.widget.Spinner;
10   import android.widget.TextView;
11   
12   import androidx.appcompat.app.AppCompatActivity;
13   
14   
15   public class CalorieCalculatorActivity extends AppCompatActivity {
16   
17       EditText ageInput, weightInput, heightInput;
18       RadioGroup genderGroup;
19       Spinner activityLevelSpinner;
20       Button calculateCaloriesButton;
21       TextView calorieResultText;
22   
23       Button seeRecipesButton;
24   
25   
26       String[] activityLevels = {
27               "Brak aktywności",
28               "Lekka aktywność",
29               "Średnia aktywność",
30               "Duża aktywność",
31               "Bardzo duża aktywność"
32       };
33       double[] activityMultipliers = {1.2, 1.375, 1.55, 1.725, 1.9};
34   
35       double result = 0;
36   
37       @Override
38       protected void onCreate(Bundle savedInstanceState) {
39           super.onCreate(savedInstanceState);
40           setContentView(R.layout.activity_calorie_calculator);
41   
42           ageInput = findViewById(R.id.ageInput);
43           weightInput = findViewById(R.id.weightInput);
44           heightInput = findViewById(R.id.heightInput);
45           genderGroup = findViewById(R.id.genderGroup);
46           activityLevelSpinner = findViewById(R.id.activityLevelSpinner);
47           calculateCaloriesButton = findViewById(R.id.calculateCaloriesButton);
48           calorieResultText = findViewById(R.id.calorieResultText);
49           seeRecipesButton = findViewById(R.id.seeRecipesButton);
50   
51           ArrayAdapter<String> adapter = new ArrayAdapter<>(this, android.R.layout.simple_spinner_dropdown_item, activityLevels);
52           activityLevelSpinner.setAdapter(adapter);
53   
54   
55           calculateCaloriesButton.setOnClickListener(v -> {
56               String ageStr = ageInput.getText().toString();
57               String weightStr = weightInput.getText().toString();
58               String heighStr = weightInput.getText().toString();
59   
60               if (ageStr.isEmpty() || weightStr.isEmpty() || heighStr.isEmpty() || genderGroup.getCheckedRadioButtonId()==-1){
61                   calorieResultText.setText("Uzupełni dane");
62                   return;
63               }
64   
65   
66               int age = Integer.parseInt(ageStr);
67               float weight = Float.parseFloat(weightStr);
68               float height = Float.parseFloat(heighStr);
69               boolean isMale = genderGroup.getCheckedRadioButtonId() == R.id.radioMale;
70   
71               double bmr;
72               if (isMale){
73                   bmr = 66.47 + (13.75 * weight) + (5.003 * height) - (6.755 * age);
74               } else {
75                   bmr = 655.1 + (9.563 * weight) + (1.850 * height) - (4.676 * age);
76               }
77               int selectedIndex = activityLevelSpinner.getSelectedItemPosition();
78               double multiplier = activityMultipliers[selectedIndex];
79   
80               result = bmr * multiplier;
81   
82               calorieResultText.setText(String.format("Twoje dzienne zapotrzebowanie: %.0f kcal", result));
83   
84   
85           });
86           seeRecipesButton.setOnClickListener(v -> {
87               Intent intent = new Intent(CalorieCalculatorActivity.this, RecipeActivity.class);
88               intent.putExtra("calories", result);
89               startActivity(intent);
90           });
91       }
92   }
